# ========== FUNCIÓN DE NORMALIZACIÓN DE NÚMEROS ==========
normalize_num() {
    local val="$1"
    # Eliminar cualquier cosa que no sea número, punto o signo menos
    val=$(echo "$val" | tr ',' '.' | sed 's/[^0-9.-]//g')
    # Si está vacío o es inválido, retornar 0
    if [[ -z "$val" ]] || [[ ! "$val" =~ ^-?[0-9]*\.?[0-9]+$ ]]; then
        echo "0"
    else
        # Usar printf para garantizar formato correcto
        printf "%.6f" "$val" 2>/dev/null || echo "0"
    fi
}

# ========== FUNCIONES DE VALIDACIÓN ==========
validar_ip() { [[ $1 =~ ^[0-9]{1,3}\.[0-9]{1,3}\.[0-9]{1,3}\.[0-9]{1,3}$ ]]; }
validar_num() { [[ $1 =~ ^[0-9]+$ ]]; }
validar_bw() { [[ $1 =~ ^[0-9]+[KMG]$ ]]; }

# ========== FUNCIÓN DE CONFIGURACIÓN INTERACTIVA ==========
configurar() {
    clear
    echo "═══════════════════════════════════════════════════════"
    echo "  COOPELESCA R.L. - Monitor de Red v15.0"
    echo "═══════════════════════════════════════════════════════"
    echo ""
    echo "Presiona ENTER para usar valor por defecto [entre corchetes]"
    echo ""

    while true; do
        read -p "IP Servidor iperf3 [$DEFAULT_SERVER]: " SERVER
        SERVER="${SERVER:-$DEFAULT_SERVER}"
        validar_ip "$SERVER" && break
        echo "  ❌ IP inválida"
    done

    while true; do
        read -p "Puerto [$DEFAULT_PORT]: " PORT
        PORT="${PORT:-$DEFAULT_PORT}"
        validar_num "$PORT" && [ "$PORT" -ge 1 ] && [ "$PORT" -le 65535 ] && break
        echo "  ❌ Puerto inválido"
    done

    while true; do
        read -p "IP para Ping [$DEFAULT_PING]: " PING_HOST
        PING_HOST="${PING_HOST:-$DEFAULT_PING}"
        validar_ip "$PING_HOST" && break
        echo "  ❌ IP inválida"
    done

    echo ""
    echo "Protocolo:"
    echo "  1) Solo TCP"
    echo "  2) Solo UDP"
    echo "  3) Ambos (RECOMENDADO)"
    while true; do
        read -p "Selecciona [3]: " PROTO
        PROTO="${PROTO:-$DEFAULT_PROTO}"
        case $PROTO in
            1) PROTO="TCP"; break ;;
            2)
                PROTO="UDP"
                while true; do
                    read -p "Bandwidth UDP (ej: 5G, 1G, 500M) [$DEFAULT_UDP_BW]: " UDP_BW
                    UDP_BW="${UDP_BW:-$DEFAULT_UDP_BW}"
                    validar_bw "$UDP_BW" && break
                    echo "  ❌ Formato inválido"
                done
                break
                ;;
            3)
                PROTO="AMBOS"
                while true; do
                    read -p "Bandwidth UDP (ej: 5G, 1G, 500M) [$DEFAULT_UDP_BW]: " UDP_BW
                    UDP_BW="${UDP_BW:-$DEFAULT_UDP_BW}"
                    validar_bw "$UDP_BW" && break
                    echo "  ❌ Formato inválido"
                done
                break
                ;;
            *) echo "  ❌ Opción inválida" ;;
        esac
    done

    echo ""
    echo "Duración del Monitoreo:"
    while true; do
        read -p "Días [$DEFAULT_DIAS]: " DIAS
        DIAS="${DIAS:-$DEFAULT_DIAS}"
        validar_num "$DIAS" && break
        echo "  ❌ Número inválido"
    done

    while true; do
        read -p "Horas (0-23) [$DEFAULT_HORAS]: " HORAS
        HORAS="${HORAS:-$DEFAULT_HORAS}"
        validar_num "$HORAS" && [ "$HORAS" -le 23 ] && break
        echo "  ❌ Horas inválidas"
    done

    while true; do
        read -p "Minutos (0-59) [$DEFAULT_MINUTOS]: " MINUTOS
        MINUTOS="${MINUTOS:-$DEFAULT_MINUTOS}"
        validar_num "$MINUTOS" && [ "$MINUTOS" -le 59 ] && break
        echo "  ❌ Minutos inválidos"
    done

    echo ""
    echo "Intervalo entre Pruebas:"
    while true; do
        read -p "Minutos (0-59) [$DEFAULT_INT_MIN]: " INT_MIN
        INT_MIN="${INT_MIN:-$DEFAULT_INT_MIN}"
        validar_num "$INT_MIN" && [ "$INT_MIN" -le 59 ] && break
        echo "  ❌ Minutos inválidos"
    done

    while true; do
        read -p "Segundos (0-59) [$DEFAULT_INT_SEC]: " INT_SEC
        INT_SEC="${INT_SEC:-$DEFAULT_INT_SEC}"
        validar_num "$INT_SEC" && [ "$INT_SEC" -le 59 ] && break
        echo "  ❌ Segundos inválidos"
    done

    INTERVALO=$((INT_MIN * 60 + INT_SEC))
    [ "$INTERVALO" -lt 1 ] && INTERVALO=30

    while true; do
        read -p "Duración de cada test en segundos [$DEFAULT_TEST_DUR]: " TEST_DUR
        TEST_DUR="${TEST_DUR:-$DEFAULT_TEST_DUR}"
        validar_num "$TEST_DUR" && [ "$TEST_DUR" -ge 1 ] && break
        echo "  ❌ Duración inválida"
    done

    local total_seg=$((DIAS * 86400 + HORAS * 3600 + MINUTOS * 60))
    local num_pruebas=1
    [ "$total_seg" -gt 0 ] && num_pruebas=$((total_seg / INTERVALO))

    echo ""
    echo "═══════════════════════════════════════════════════════"
    echo "RESUMEN DE CONFIGURACIÓN:"
    echo "═══════════════════════════════════════════════════════"
    echo "Servidor iperf3: $SERVER:$PORT"
    echo "Ping a: $PING_HOST"
    echo "Protocolo: $PROTO"
    [ "$PROTO" != "TCP" ] && echo "UDP Bandwidth: $UDP_BW"
    echo "Duración total: $DIAS días, $HORAS horas, $MINUTOS minutos"
    echo "Intervalo: $INTERVALO segundos"
    echo "Duración test: $TEST_DUR segundos"
    echo ""
    echo "📊 Se realizarán aproximadamente $num_pruebas pruebas"
    echo "═══════════════════════════════════════════════════════"
    echo ""
    read -p "¿Iniciar monitoreo? (S/n): " confirm
    confirm="${confirm:-S}"
    [[ ! $confirm =~ ^[Ss]$ ]] && { echo "Cancelado"; exit 0; }
}

# ========== VERIFICACIÓN DE HERRAMIENTAS ==========
check_tools() {
    local missing=""
    if ! command -v iperf3 >/dev/null 2>&1; then
        echo "❌ iperf3 no está instalado"
        missing="iperf3 "
    fi
    if ! command -v jq >/dev/null 2>&1; then

    fi
    if [ -n "$missing" ]; then
        echo "   Instala con: sudo apt-get install $missing"
        exit 1
    fi
}

# ========== FUNCIONES DE MEDICIÓN TCP ==========
tcp_dl() {
    local tmp="/tmp/tcp_dl_$$"
    if iperf3 -c "$SERVER" -p "$PORT" -R -t "$TEST_DUR" -J > "$tmp" 2>&1; then
        local bps=$(jq -r '.end.sum_received.bits_per_second // 0' "$tmp" 2>/dev/null)
        local retrans=$(jq -r '.end.sum_received.retransmits // 0' "$tmp" 2>/dev/null)
        rm -f "$tmp"

        bps=$(normalize_num "$bps")
        retrans=$(normalize_num "$retrans")
        local mbps=$(awk -v b="$bps" 'BEGIN{printf "%.2f", b/1000000}')

        echo "${mbps}:${retrans}"
    else
        rm -f "$tmp"
        echo "0:0"
    fi
}

}

# ========== FUNCIONES DE MEDICIÓN UDP ==========
udp_dl() {
    local tmp="/tmp/udp_dl_$$"
    if iperf3 -c "$SERVER" -p "$PORT" -R -u -b "$UDP_BW" -t "$TEST_DUR" -J > "$tmp" 2>&1; then
        local bps=$(jq -r '.end.sum.bits_per_second // 0' "$tmp" 2>/dev/null)
        local jit=$(jq -r '.end.sum.jitter_ms // 0' "$tmp" 2>/dev/null)
        local loss=$(jq -r '.end.sum.lost_percent // 0' "$tmp" 2>/dev/null)
        rm -f "$tmp"

        bps=$(normalize_num "$bps")
        jit=$(normalize_num "$jit")
        loss=$(normalize_num "$loss")
        local mbps=$(awk -v b="$bps" 'BEGIN{printf "%.2f", b/1000000}')

        echo "${mbps}:${jit}:${loss}"
    else
        rm -f "$tmp"
        echo "0:0:0"
    fi
}

# ========== FUNCIÓN DE MEDICIÓN PING ==========
ping_test() {
    local result=$(ping -c 5 -W 2 "$PING_HOST" 2>&1)
    if echo "$result" | grep -q "rtt"; then
        local avg=$(echo "$result" | awk -F'/' '/rtt/{print $5}')
        local loss=$(echo "$result" | awk -F', ' '/packet loss/{print $3}' | sed 's/%.*//')
        avg=$(normalize_num "$avg")
        loss=$(normalize_num "$loss")
        echo "${avg}:${loss}"
    else
        echo "0:100"
    fi
}

    const r = parseInt(hex.slice(1, 3), 16);

}

function renderCharts() {

        document.body.innerHTML = '<h1 style="text-align:center;margin-top:50px;color:white;">Sin datos para mostrar</h1>';

    Object.keys(charts).forEach(k => totals[k] = 0);

    dataPoints.forEach(data => {
        const timeLabel = new Date(data.timestamp).toLocaleTimeString();

        Object.keys(charts).forEach(chartKey => {
            if (data[chartKey] !== undefined && data[chartKey] !== null) {
                charts[chartKey].data.labels.push(timeLabel);
                const val = parseFloat(data[chartKey]) || 0;

        });

# ========== FUNCIÓN DE FINALIZACIÓN DEL HTML ==========
finalizar_html() {
    local inicio="$1"
    local fin="$2"

    sed -i "s|SERVIDOR_PLACEHOLDER|$SERVER:$PORT|g" "$HTML"
    sed -i "s|PING_HOST_PLACEHOLDER|$PING_HOST|g" "$HTML"
    sed -i "s|PROTOCOLO_PLACEHOLDER|$PROTO|g" "$HTML"
    sed -i "s|INICIO_PLACEHOLDER|$inicio|g" "$HTML"
    sed -i "s|FIN_PLACEHOLDER|$fin|g" "$HTML"
    sed -i "s|INTERVALO_PLACEHOLDER|$INTERVALO|g" "$HTML"

    if [ -f "$DATA" ] && [ -s "$DATA" ]; then
        sed -i "/\/\/ DATA_PLACEHOLDER/r $DATA" "$HTML"
    fi
    sed -i "/\/\/ DATA_PLACEHOLDER/d" "$HTML"

    rm -f "$DATA"
}
